137inline std::pair<RectangleTileMap, Coords> loadLevelMap(const std::filesystem::path& path,
138 const GameObjectFactory& objectFactory)
139{
140 auto file = std::ifstream { path };
141 if (!file.is_open()) {
142 throw std::runtime_error("Failed to open path " + path.string());
143 }
144
145 static const std::string symbolsToIgnore = "\r";
146 std::vector<RectangleTileMap::Row> map;
147 std::optional<Coords> playerCoords_;
148 std::string line;
149 std::size_t rowsCount = 0;
150 while (std::getline(file, line)) {
151 RectangleTileMap::Row row(line.size());
152 for (std::size_t i = 0; i < line.size(); ++i) {
153 if (symbolsToIgnore.find(line[i]) != std::string::npos) {
154 continue;
155 }
156 auto objects = objectFactory.create(line[i]);
157 for (const auto& object : objects) {
158 if (*object->id() == "player") {
159 playerCoords_ = { i, rowsCount };
160 }
161 }
162 row[i].setObjects(std::move(objects));
163 }
164 map.push_back(std::move(row));
165 ++rowsCount;
166 }
167
168 return { RectangleTileMap { std::move(map) }, playerCoords_.value() };
169}
